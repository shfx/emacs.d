1 Emacs Initial config

1.1 Initialiation defaults

1.1.1 Setting default coding system

(set-language-environment "UTF-8")
(set-default-coding-systems 'utf-8)
(set-terminal-coding-system 'utf-8)
(set-keyboard-coding-system 'utf-8)

1.1.2 Setting default display and editing options

(set-default 'truncate-lines t)
(global-hl-line-mode 0)
(set-window-margins nil 2 2)
(electric-pair-mode 1)
(line-number-mode 0)
(column-number-mode 0)
(global-linum-mode 0)
(setq inhibit-splash-screen t
      inhibit-startup-message t
      pop-up-frames nil
      visible-bell 'top-bottom
      ring-bell-function 'ignore
      transient-mark-mode t
      show-paren-mode 1
      make-backup-files nil
      auto-save-default nil)

1.1.3 Set the titlebar to the color of the theme

(when (memq window-system '(mac ns))
  (add-to-list 'default-frame-alist '(ns-appearance . light))
  (add-to-list 'default-frame-alist '(ns-transparent-titlebar . t)))

1.1.4 Setting title format

(setq frame-title-format
  '("Emacs - " (buffer-file-name "%f"
  ' (dired-directory dired-directory "%b"))))

1.1.5 Seting up package system and bootstrapping use-package

(require 'package)

(add-to-list 'package-archives '("melpa" . "http://melpa.org/packages/"))
(add-to-list 'package-archives '("org" . "https://orgmode.org/elpa/") t)

(when (< emacs-major-version 24)
  (add-to-list 'package-archives
               '("gnu" . "http://elpa.gnu.org/packages/")))

(package-initialize)

(unless (package-installed-p 'use-package)
  (package-refresh-contents)
  (package-install 'use-package))

1.1.6 System specyfic defaults

(when (eq system-type 'darwin)
  (defvar ls-lisp-use-insert-directory-program)
  (defvar powerline-image-apple-rgb)
  (setq ns-use-srgb-colorspace t)
  (setq powerline-image-apple-rgb t)
  (require 'ls-lisp)
  (setq ls-lisp-use-insert-directory-program nil))

1.2 Packages

1.2.1 Mac Only - initialize emacs with proper PATH

(use-package exec-path-from-shell
  :if (memq window-system '(mac ns))
  :ensure t
  :config
  (exec-path-from-shell-initialize)
  (exec-path-from-shell-copy-env "GOPATH"))

1.2.2 Load theme

(use-package doom-themes
  :ensure t
  :init
  (load-theme 'doom-molokai t))
